async function deployToHF(){
  const token=document.getElementById('hfToken').value.trim();
  const spaceName=document.getElementById('hfSpace').value.trim().replace(/[^a-zA-Z0-9\-_]/g,'-');
  const isPrivate=document.getElementById('hfPrivate').value==='true';
  const license=document.getElementById('hfLicense').value;
  if(!token||!spaceName){hfLog('❌ Token en Space naam zijn verplicht','err');return}
  localStorage.setItem('viz-hf-token',token);
  localStorage.setItem('viz-hf-space',spaceName);
  document.getElementById('hfLog').innerHTML='';
  const content=currentLang==='html'?getFullHTML():editor.value;
  // Step 1: Get username
  hfLog('📡 Connecting to Hugging Face...','info');
  let username='';
  try{
    const whoRes=await fetch('https://huggingface.co/api/whoami-v2',{
      headers:{'Authorization':'Bearer '+token}
    });
    if(!whoRes.ok)throw new Error('Invalid token (HTTP '+whoRes.status+')');
    const whoData=await whoRes.json();
    username=whoData.name;
    hfLog('✅ Authenticated as: '+username,'ok');
  }catch(e){
    hfLog('❌ Auth failed: '+e.message,'err');
    return;
  }
  const repoId=username+'/'+spaceName;
  // Step 2: Create Space (if not exists)
  hfLog('📦 Creating Space: '+repoId+'...','info');
  try{
    const createRes=await fetch('https://huggingface.co/api/repos/create',{
      method:'POST',
      headers:{'Authorization':'Bearer '+token,'Content-Type':'application/json'},
      body:JSON.stringify({
        type:'space',
        name:spaceName,
        sdk:'static',
        private:isPrivate,
        license:license
      })
    });
    if(createRes.ok){
      hfLog('✅ Space created!','ok');
    }else{
      const err=await createRes.json().catch(()=>({}));
      if(err.error&&err.error.includes('already')){
        hfLog('ℹ️ Space already exists, updating...','info');
      }else{
        hfLog('⚠️ Create response: '+(err.error||err.message||createRes.status),'info');
      }
    }
  }catch(e){
    hfLog('⚠️ Create step: '+e.message+' (may already exist)','info');
  }
  // Step 3: Upload index.html via commit API
  hfLog('⬆️ Uploading index.html ('+content.length+' chars)...','info');
  try{
    // Use the upload_file API
    const blob=new Blob([content],{type:'text/plain'});
    const formData=new FormData();
    formData.append('file',blob,'index.html');
    // First try the simple upload endpoint
    const uploadRes=await fetch('https://huggingface.co/api/spaces/'+repoId+'/upload/main/index.html',{
      method:'PUT',
      headers:{'Authorization':'Bearer '+token},
      body:blob
    });
    if(uploadRes.ok){
      hfLog('✅ index.html uploaded!','ok');
    }else{
      // Fallback: use commit API
      hfLog('ℹ️ Trying commit API...','info');
      const commitPayload={
        summary:'Deploy from HTML Visualizer',
        files:[{path:'index.html',content:content}]
      };
      // Use LFS-free commit
      const boundary='----HFBoundary'+Date.now();
      const header_part='--'+boundary+'\r\nContent-Disposition: form-data; name="header"\r\nContent-Type: application/json\r\n\r\n';
      const headerJson=JSON.stringify({summary:'Deploy from HTML Visualizer',parentCommit:undefined});
      const file_part='\r\n--'+boundary+'\r\nContent-Disposition: form-data; name="file"; filename="index.html"\r\nContent-Type: text/html\r\n\r\n';
      const end='\r\n--'+boundary+'--';
      const bodyStr=header_part+headerJson+file_part+content+end;
      const commitRes=await fetch('https://huggingface.co/api/spaces/'+repoId+'/commit/main',{
        method:'POST',
        headers:{
          'Authorization':'Bearer '+token,
          'Content-Type':'multipart/form-data; boundary='+boundary
        },
        body:bodyStr
      });
      if(!commitRes.ok){
        const errData=await commitRes.text();
        throw new Error('Upload failed: '+commitRes.status+' '+errData.substring(0,200));
      }
      hfLog('✅ Committed successfully!','ok');
    }
  }catch(e){
    hfLog('❌ Upload error: '+e.message,'err');
    hfLog('💡 Tip: zorg dat je token "write" permissions heeft','info');
    return;
  }
  // Step 4: Show live URL
  const liveUrl='https://'+username+'-'+spaceName+'.hf.space';
  const spaceUrl='https://huggingface.co/spaces/'+repoId;
  hfLog('','ok');
  hfLog('🎉 Deployment complete!','ok');
  hfLog('🌐 Live URL: '+liveUrl+'','ok');
  hfLog('📋 Space: '+spaceUrl+'','ok');
  hfLog('','info');
  hfLog('⏳ Het kan 1-2 minuten duren voordat de Space live is.','info');
}

let lastAgentOutput='';
async function sendToAgent(){
  const apiKey=document.getElementById('agptKey').value.trim();
  let agentId=document.getElementById('agentSelect').value;
  if(agentId==='custom')agentId=document.getElementById('customAgentId').value.trim();
  const prompt=document.getElementById('agentPrompt').value.trim();
  if(!apiKey){agentLog('❌ API key required','err');return}
  if(!agentId){agentLog('❌ Agent ID required','err');return}
  localStorage.setItem('viz-agpt-key',apiKey);
  const code=currentLang==='html'?getFullHTML():editor.value;
  const fullPrompt=prompt+'\n\nLanguage: '+currentLang+'\n\n```\n'+code+'\n```';
  document.getElementById('agentLog').innerHTML='';
  document.getElementById('agentResult').style.display='none';
  document.getElementById('sendAgentBtn').textContent='⏳ Sending...';
  document.getElementById('sendAgentBtn').disabled=true;
  agentLog('📡 Connecting to AutoGPT API...','info');
  agentLog('🤖 Agent: '+agentId,'info');
  try{
    // Trigger agent execution via AutoGPT External API
    agentLog('⬆️ Sending code ('+code.length+' chars)...','info');
    const execRes=await fetch('https://backend.agpt.co/api/v2/library/agents/'+agentId+'/execute',{
      method:'POST',
      headers:{'Authorization':'Bearer '+apiKey,'Content-Type':'application/json'},
      body:JSON.stringify({input_data:{input:fullPrompt}})
    });
    if(!execRes.ok){
      const err=await execRes.json().catch(()=>({detail:'HTTP '+execRes.status}));
      throw new Error(err.detail||err.message||'Execute failed: '+execRes.status);
    }
    const execData=await execRes.json();
    const execId=execData.id||execData.execution_id;
    agentLog('✅ Execution started: '+execId,'ok');
    agentLog('⏳ Waiting for result...','info');
    // Poll for result
    let attempts=0;
    const maxAttempts=60;
    const poll=async()=>{
      attempts++;
      try{
        const res=await fetch('https://backend.agpt.co/api/v2/library/agents/'+agentId+'/executions/'+execId,{
          headers:{'Authorization':'Bearer '+apiKey}
        });
        if(res.ok){
          const data=await res.json();
          if(data.status==='completed'||data.status==='COMPLETED'){
            agentLog('✅ Agent completed!','ok');
            // Extract output
            let output='';
            if(data.output_data){
              if(typeof data.output_data==='string')output=data.output_data;
              else if(data.output_data.output)output=data.output_data.output;
              else if(data.output_data.response)output=data.output_data.response;
              else output=JSON.stringify(data.output_data,null,2);
            }else if(data.output){
              output=typeof data.output==='string'?data.output:JSON.stringify(data.output,null,2);
            }else{
              output=JSON.stringify(data,null,2);
            }
            lastAgentOutput=output;
            document.getElementById('agentOutput').textContent=output;
            document.getElementById('agentResult').style.display='block';
            document.getElementById('sendAgentBtn').textContent='🤖 Send to Agent';
            document.getElementById('sendAgentBtn').disabled=false;
            return;
          }else if(data.status==='failed'||data.status==='FAILED'){
            throw new Error('Agent execution failed: '+(data.error||'unknown error'));
          }else{
            agentLog('⏳ Status: '+data.status+' (attempt '+attempts+'/'+maxAttempts+')','info');
          }
        }
      }catch(e){
        if(e.message.includes('failed')){throw e}
      }
      if(attempts>=maxAttempts){
        agentLog('⏰ Timeout — check AutoGPT dashboard for results','err');
        document.getElementById('sendAgentBtn').textContent='🤖 Send to Agent';
        document.getElementById('sendAgentBtn').disabled=false;
        return;
      }
      setTimeout(poll,5000);
    };
    setTimeout(poll,3000);
  }catch(e){
    agentLog('❌ Error: '+e.message,'err');
    document.getElementById('sendAgentBtn').textContent='🤖 Send to Agent';
    document.getElementById('sendAgentBtn').disabled=false;
  }
}
function applyAgentResult(){
  // Try to extract code block from agent output
  let code=lastAgentOutput;
  const codeMatch=code.match(/```(?:\w+)?\n([\s\S]*?)```/);
  if(codeMatch)code=codeMatch[1].trim();
  editor.value=code;
  autoSave();
  updatePreview();
  closeAgent();
  addConsoleLine('log',['🤖 Agent result applied to editor']);
}
function copyAgentResult(){
  navigator.clipboard.writeText(lastAgentOutput);
  event.target.textContent='✅ Copied!';
  setTimeout(()=>event.target.textContent='📋 Copy',1500);
}

function expandEmmet(){
  const sel=editor.selectionStart;
  const before=editor.value.substring(0,sel);
  const after=editor.value.substring(sel);
  // Find the abbreviation (last word before cursor)
  const match=before.match(/([\.#\w\[\]\>\+\*\(\)\{\}\^]+)$/);
  if(!match){addConsoleLine('warn',['No Emmet abbreviation found at cursor']);return}
  const abbr=match[1];
  const expanded=emmetExpand(abbr);
  if(expanded===abbr){addConsoleLine('warn',['Unknown abbreviation: '+abbr]);return}
  editor.value=before.substring(0,before.length-abbr.length)+expanded+after;
  editor.selectionStart=editor.selectionEnd=before.length-abbr.length+expanded.length;
  autoSave();updatePreview();
  addConsoleLine('log',['Emmet expanded: '+abbr+' → ('+expanded.length+' chars)']);
}
